\chapter{Methodology}\label{meth}
This chapter presents the methodology used in this thesis to train the neural control policies.
In~\autoref{sec:quad_model}, we define the quadrotor model and in \autoref{sec:ca} we present the four different control abstractions that have been implemented.
Next, in \autoref{sec:DR} we explain how the \ac{s2r} gap is minimized by using \ac{DR}.
\autoref{sec:tasks}, outlines the training objectives used to address the \ac{ADR} challenge.
Lastly in \autoref{sec:summary}, we summarize the learning framework  and we present our final training algorithm.



\section{Quadrotor Model}\label{sec:quad_model}
In the simulated environments, the quadcopter is modeled as a 6\ac{dof} rigid body.
The inertial world frame $\mathcal{W}$ is defined by the fixed orthonormal basis $\{\hat{\mathbf{e}}_1, \hat{\mathbf{e}}_2, \hat{\mathbf{e}}_3\}$, while the non-inertial body frame $\mathcal{B}$ is defined by the orthonormal basis $\{\hat{\mathbf{b}}_1, \hat{\mathbf{b}}_2, \hat{\mathbf{b}}_3\}$, which is attached to the vehicle's center of mass.  
The state of the \ac{UAV} is given by $\mathbf{x} = [\mathbf{p}, \mathbf{v}, \mathbf{q}, \boldsymbol{\omega}, \mathbf{\Omega}]^\top$, where the position $\mathbf{p} \in \real^3$ and velocity $\mathbf{v} \in \real^3$ are represented in the world frame $\mathcal{W}$. 
Conversely, the body rates $\boldsymbol{\omega} \in \real^3$ are represented in the body frame $\mathcal{B}$, and $\mathbf{\Omega} \in \real^4$ is a vector of scalar values indicating the rotor speeds.
The relationship between the two frames is given by the rotation mapping $\mathbf{b}_i = \mathbf{R}\mathbf{e}_i$, where $\mathbf{R} \in \mathrm{SO}(3)$.

The kinematics and dynamics of the quadcopter can be described by the following set of equations:

\begin{minipage}{0.4\textwidth}
\begin{equation}
  \dot{\mathbf{p}}^{\mathcal{W}} = \mathbf{v}^{\mathcal{W}}
\end{equation}
\begin{equation}
    \dot{\mathbf{q}}^{\mathcal{W}} = \tfrac{1}{2}\ \mathbf{q}^{\mathcal{W}} \odot \begin{bmatrix} 0 \\ \boldsymbol{\omega}^{\mathcal{B}} \end{bmatrix}
\end{equation}
\begin{equation}
    \dot{\mathbf{v}}^{\mathcal{W}} = \frac{\mathbf{R}(\mathbf{q})}{m} \left( \mathbf{F}_T^{\mathcal{B}} + \mathbf{F}_D^{\mathcal{B}} \right) + \mathbf{g}^{\mathcal{W}}
\end{equation}
\end{minipage}%
\begin{minipage}{0.5\textwidth}
\begin{equation}
    \dot{\boldsymbol{\omega}}^{\mathcal{B}} = \mathbf{J}^{-1} \left( \boldsymbol{\tau}^{\mathcal{B}} - \boldsymbol{\omega}^{\mathcal{B}} \times \mathbf{J} \boldsymbol{\omega}^{\mathcal{B}} \right)
\end{equation}
\begin{equation}
  \dot{\mathbf{\Omega}} = \frac{1}{\tau_{\mathrm{m}}} \left(\mathbf{\Omega}_{\text{cmd}} - \mathbf{\Omega}\right)
\end{equation}
\end{minipage}

\noindent In these expressions, $\odot$ denotes the quaternion multiplication operator, $\mathbf{R}(\mathbf{q})$ represents the rotation matrix mapping the body frame $\mathcal{B}$ to the world frame $\mathcal{W}$.
The vectors $\mathbf{F}_T^{\mathcal{B}}$ and $\mathbf{F}_D^{\mathcal{B}}$ correspond to the collective thrust and drag forces, respectively.
Finally, $\mathbf{J}$ is the diagonal inertia matrix, $\tau_m$ is the motor time constant, and $\mathbf{g}^{\mathcal{W}}$ is the gravity vector.

In the body frame $\mathcal{B}$, the individual motor thrusts $T_i, \ i=1,..,4$, are assumed to be perpendicular to the $\{\hat{\mathbf{b}}_1,\hat{\mathbf{b}}_2\}$ plane.
To be strictly rigorous, the torques produced by the rotors' angular acceleration and gyroscopic effects should be accounted for; nonetheless, these effects can be neglected~\cite{sun_comparative_2024}.
Hence, the mapping from motor thrusts to collective thrust $F_T$, and body torques $\boldsymbol{\tau}^\mathcal{B}$ can be obtained from \eqref{eq:simp_alloc}, (see \autoref{f-diag}).

\begin{equation}
  \begin{bmatrix}
    F_T \\
    \tau_{b1} \\
    \tau_{b2} \\
    \tau_{b3}
  \end{bmatrix} = 
  \overbrace{
  \begin{bmatrix}
    1 & 1 & 1 & 1 \\
    -l\sin{\beta} & l\sin{\beta} & l\sin{\beta} & -l\sin{\beta} \\
    -l\cos{\beta} & l\cos{\beta} & -l\cos{\beta} & l\cos{\beta} \\
    -c_{tf} & -c_{tf} & c_{tf} & c_{tf}
  \end{bmatrix}
  }^{\boldsymbol{\Gamma}}
  \begin{bmatrix}
    T_1 \\ T_2 \\ T_3 \\ T_4
  \end{bmatrix}
  \label{eq:simp_alloc}
\end{equation}
Where $l$ is the arm length, $\beta$ is the arm angle, and $c_{tf}$ the linear torque constant.

The individual motror thrusts $T_i$, can be obtained from the rotor speeds $\Omega_i$ with the thrust map coefficents, $\mathbf{k_T} = [a_1,\ a_2,\ a_3]$ which can be obtained experimentally by fitting a quadratic curve to the rotor speed versus thrust data.

\begin{equation} 
  T_i = \mathbf{k_T} \begin{bmatrix}
    \Omega_i^2 \\ \Omega_i \\ 1
  \end{bmatrix} = \begin{bmatrix}
  a_0 & a_1 & a_2
  \end{bmatrix} \begin{bmatrix}
    \Omega_i^2 \\ \Omega_i \\ 1
  \end{bmatrix}
    \label{eq:thrust_map}, \quad i = 1,\dots,4
\end{equation}
The terms $a_1$ and $a_2$ can be neglected, hence \eqref{eq:thrust_map} is simplified to $ T_i =  a_0 \Omega_i^2$. 


To complete the quadrotor model, the aerodynamic drag force $\mathbf{F}^{\mathcal{B}}_D$ is incorporated using a simplified quadratic formulation.
Other aerodynamic effects, such as rotational drag and ground interaction, are neglected.
The drag force is expressed as,
\begin{equation}
    \mathbf{F}^{\mathcal{B}}_D = -\tfrac{1}{2}\rho \|\mathbf{v}^{\mathcal{B}}\| \mathbf{C}_D \mathbf{A} \mathbf{v}^{\mathcal{B}}
\end{equation}
where $\rho$ denotes the air density, $\mathbf{A} = \mathrm{diag}(A_{b1}, A_{b2}, A_{b3})$ represents the reference area along each body axis, and $\mathbf{C_D} = \mathrm{diag}(C_{D_{b1}}, C_{D_{b2}}, C_{D_{b3}})$  is a diagonal matrix of non-dimensional drag coefficients.
The velocity must be expressed in the body frame, obtained as $\mathbf{v}^{\mathcal{B}} = \mathbf{R}(\mathbf{q})^\top \mathbf{v}^{\mathcal{W}}$.


\section{Control Abstractions}\label{sec:ca}
In this thesis, we refer to the control abstraction or control mode as the modality of the policy outputs used to control the quadrotor.
The choice of control abstraction affects both controller performance for a given task and sample efficiency during training.
For this reason, it is worthwhile to investigate and implement multiple control abstractions.

In this thesis, three control modalities are considered.
For all control modalities, the policy output layer passes through a sigmoid activation function, \eqref{sigmoid}, ensuring that the action space is bounded in $[0,\ 1]$.
\begin{equation}\label{sigmoid}
    \sigma(x) = \frac{1}{1 + e^{-x}}
\end{equation}

The lowest level of abstraction consists of directly outputting \ac{SRT}.
Then, we implement an intermediate level of abstraction that outputs \ac{CTBR}, which are subsequently mapped to individual rotor thrusts through the control allocation matrix \eqref{eq:simp_alloc}.
The highest level of abstraction outputs \ac{LVHR}, which are converted into wrench using an $\mathrm{SO}(3)$ geometric controller, inspired by~\cite{lee_geometric_2010}, and subsequently mapped to rotor thrusts in an analogous manner to the \ac{CTBR} formulation.
Next, we explain in more detail our implementation of these controllers.

\subsection{Single Rotor Thrust} \label{sec:srt}
When the \ac{SRT} control mode is chosen, the only transformation applied is scaling the policy output $\boldsymbol{u} \in [0, 1]\in \real^4$ to the actuator limits, $\mathbf{T_{\text{cmd}}} \in [T_{\text{min}}, T_{\text{max}} ] \in \real^4$.
Therefore, the transformation is as simple as,

\begin{equation}
    T_{\text{cmd},i} = u_i \left( T_{\text{max}}- T_{\text{min}} \right) + T_{\text{min}}, \quad i = 1, \dots, 4.
\end{equation}


Once the scaling is applied, a first-order model with rotor speed saturation is implemented to better emulate the behaviour of a real motor.
First, the commanded motor thrusts $\mathbf{T}_{\text{cmd}}$ are converted to rotor speeds with,

\begin{equation} \label{eq:motspd}
    \mathbf{T}_{\text{cmd}} \approx k \boldsymbol{\Omega}_{\text{cmd}}^2.
\end{equation}

Then the commanded rotor speeds $\boldsymbol{\Omega}_{\text{cmd}}$ are passed through a first-order transfer function, \eqref{eq:motmod}, and saturated according to the maximum rotor speed rate, $\dot{\Omega}_{\text{cmd}}$.

\begin{equation}\label{eq:motmod}
    H(s) = \frac{1}{\tau_m s + 1}
\end{equation}
where $\tau_m$ is the motor time constant.
Finally, the rotor speeds obtained after the described process are mapped again to \ac{SRT} values using \eqref{eq:motspd} and then converted to \ac{CTBR} with \eqref{eq:simp_alloc} to integrate the dynamics.
Modeling the motor dynamics provides the policy with a realistic data distribution.
However, for training purposes, it has been decided to skip the motor dynamics during the backward pass, as including them significantly destabilizes gradients and complicates the optimization process.
In particular, the mapping in \eqref{eq:motspd} produces large variations in the gradient, degrading training performance and sample efficiency.

% CTBR
\subsection{Collective Thrust and Body Rates}\label{ctbr-bad}
When the \ac{CTBR} control mode is selected, the normalized outputs of the policy $\boldsymbol{u} \in [0, 1]\in \real^4$, are scaled as follows,

\begin{equation}
    T_{c, \text{cmd}} = u_1  (T_{\text{max}} - T_{\text{min}}) + T_{\text{min}} 
\end{equation}
\begin{equation}\label{omega-cmd}
    \omega_{i, \text{cmd}} = (2u_{i+1} - 1)  \omega_{i, \text{max}}, \quad i = 1, 2, 3
\end{equation}

\noindent where $\omega_{i, \text{max}}$ indicates the maximum body rate of the respective axis, (roll, pitch, yaw).
After scaling the angluar rates, they are converted to torque values:

\begin{equation}
    \boldsymbol{\tau}_\text{cmd} = k_p \frac{(\boldsymbol{\omega_{\text{cmd}}} - \boldsymbol{\omega})}{dt} J    
\end{equation}
Additionally, a proportional gain $k_p$ can be introduced to adjust the stiffness of the conversion.
Given the commanded collective thrust and torques, the motor thrusts can be obtained from \eqref{eq:simp_alloc}, and the same procedure described in \autoref{sec:srt} is applied during the forward and backward passes.

This initial formulation of the \ac{CTBR} controller is used in \autoref{sec:ca_eval}. However, in \autoref{sec:mb-vs-mf} we reformulate the \ac{CTBR} controller to better match the PixHawk 4 firmware (see \refapp{px4-mixing}).

\subsection{Linear Velocities and Heading Rate}\label{subsec:lvhr}

The highest level of abstraction that has been implemented consists of commanding \ac{LVHR}. With this control mode, the policy outputs are scaled as follows,

\begin{equation}\label{eq:geo_vel}
    v_{i, \text{cmd}} = (2u_{i} - 1)  v_{i, \text{max}}, \quad i = 1, 2, 3
\end{equation}
\begin{equation}\label{eq:geo_omega}
    \dot{\eta}_{\text{cmd}} = (2u_{4} - 1)  \dot{\eta}_{ \text{max}}
\end{equation}

\noindent where $\eta$ denotes the heading angle and $v_{i,\text{cmd}}$ represents the commanded linear velocity along each body axis. 
Mapping these control commands to thrust commands is less direct than the previous methods. 
Instead, the collective thrust and torques are obtained by applying a transformation in the Special Orthogonal group $\mathrm{SO}(3)$.
The transformations proceed in the following manner.

First, the acceleration vector is computed as,
\begin{equation}
    \mathbf{a}_{\text{cmd}} = k_v (\mathbf{v}_{\text{cmd}} - \mathbf{v}) + g \mathbf{\hat{e}}_3
\end{equation}

\noindent where $k_v$ is a proportional gain on the velocity. 
Note that the acceleration along the third body axis is compensated for gravity; $\mathbf{\hat{e}}_3$ denotes the unit vector along this axis.
Then, the force vector is the \ac{UAV}'s mass times the commanded acceleration vector, and the commanded collective thrust vector is the projection of the force vector onto the $\mathbf{\hat{b}}_3$ axis.

\begin{equation}
    \mathbf{F} = m \mathbf{a}_{\text{cmd}} 
\end{equation}

\begin{equation}\label{eq:geo_thrust}
    T_{c, \text{cmd}} = \mathbf{F} \cdot \mathbf{\hat{b}}_3
\end{equation}



To obtain the commanded body torques, the heading vector is obtained by projecting the heading angle into the world plane defined by span$\{\hat{e}_1, \hat{e}_2\}$ (see \autoref{ooo}.a).
The heading vector serves as an auxiliary vector to compute $\mathbf{\hat{b}}_2$ and $\mathbf{\hat{b}}_1$. 

\begin{equation} \mathbf{h} = \begin{bmatrix} \cos{\eta} & \sin{\eta} & 0 \end{bmatrix} ^\top \end{equation} 

\begin{equation}
    \hat{\mathbf{b}}_2 = \frac{\hat{\mathbf{b}}_3 \times \mathbf{h}}{\|\hat{\mathbf{b}}_3 \times \mathbf{h}\|}
\end{equation}

\begin{equation}
    \hat{\mathbf{b}}_1 = \hat{\mathbf{b}}_2 \times \hat{\mathbf{b}}_3
\end{equation}

\noindent In this way, the commanded orientation is defined by $\mathbf{R}_{\text{cmd}} = \begin{bmatrix}
    \hat{\mathbf{b}}_1 & \hat{\mathbf{b}}_2 & \hat{\mathbf{b}}_3
\end{bmatrix}$ and the error between the commanded orientation and the current orientation (see \autoref{ooo}.b) is computed as, 

\begin{equation}\label{eq:atterr}
    \mathbf{e}_R = \frac{1}{2} \left( \mathbf{R}_{\text{cmd}}^\top \mathbf{R} - \mathbf{R}^\top \mathbf{R}_{\text{cmd}} \right)^\vee. 
\end{equation}

\noindent The complete derivation of \eqref{eq:atterr} can be found in~\cite{lee_geometric_2010}. 
To provide intuition, note that if $\mathbf{R}_{\text{cmd}} = \mathbf{R}$, then $\mathbf{R}_{\text{cmd}}^\top \mathbf{R} = \mathbf{I}$ and the error is zero. 
Otherwise, the matrix $\mathbf{R}_{\text{cmd}}^\top \mathbf{R}$ represents the relative rotation between the current and desired orientations. 
The skew-symmetric part of the matrix  $\frac{1}{2}(\mathbf{R}_{\text{cmd}}^\top \mathbf{R} - \mathbf{R}^\top \mathbf{R}_{\text{cmd}})$, encodes the axis and magnitude of the rotation error. 
The vee map $(\cdot)^\vee$ then converts this skew-symmetric matrix into a vector in $\mathbb{R}^3$, corresponding to the rotation error expressed in the Lie algebra $\mathfrak{so}(3)$, that is to say, $^\vee \colon \mathfrak{so}(3) \to \mathbb{R}^3$.
Then, the factor $\frac{1}{2}$ ensures consistency with the standard definition of the vee operator and avoids double counting of the off-diagonal terms.



Additionally, an error in the angular velocity of the rotation is introduced as,

\begin{equation}\label{eq:geo_ang}
    \mathbf{e}_{\boldsymbol{\omega}} = \boldsymbol{\omega} - \mathbf{R}^\top \mathbf{R}_{\text{cmd}}\boldsymbol{\omega}_{\text{cmd}}
\end{equation}
where $\boldsymbol{\omega}_{\text{cmd}} = \begin{bmatrix}
    0 & 0 & \dot{\eta}_{\text{cmd}}\end{bmatrix}^\top$. 
Note that the error between the desired rotational velocity and the current rotational velocity lies on different tangent planes to the unit sphere, 
$\dot{\mathbf{R}} \in \mathrm{T}_{\mathbf{R}} \mathrm{SO}(3)$ and $\dot{\mathbf{R}}_{\text{cmd}} \in \mathrm{T}_{\mathbf{R}_{\text{cmd}}} \mathrm{SO}(3)$.
For this reason, as explained in \cite{lee_geometric_2010}, $\dot{\mathbf{R}}_{\text{cmd}}$ is transformed into a vector in the $\mathrm{T}_{\mathbf{R}} \mathrm{SO}(3)$ space to be able to compare it to $\dot{\mathbf{R}}$ and obtain the error; \eqref{eq:geo_ang}.

Finally, the torque is set by,

\begin{equation}\label{eq:geo_tau}
    \boldsymbol{\tau}_\textbf{cmd} = - k_R \mathbf{e}_\textbf{R} - k_\omega \mathbf{e}_{\boldsymbol{\omega}} + \boldsymbol{\omega}  \times \mathbf{J}\boldsymbol{\omega} - \mathbf{J} ( \hat{\boldsymbol{\omega}} \mathbf{R}^\top \mathbf{R}_{\text{cmd}} \boldsymbol{\omega}_{\text{cmd}} - \mathbf{R}^\top \mathbf{R}_{\text{cmd}} \dot{\boldsymbol{\omega}}_{\text{cmd}} )
\end{equation}
where gyroscopic and feedforward terms are added to the rotation and rotational speed errors.
The \textit{hat map} $\hat{\cdot} \colon \mathbb{R}^3 \to \mathfrak{so}(3)$ is defined by the condition that $\hat{x} y = x \times y$ for all $x, y \in \mathbb{R}^3$.
In this manner, the commanded collective thrust, \eqref{eq:geo_thrust}, and commanded torques, \eqref{eq:geo_tau}, can be mapped to single rotor thrust commands with the \ac{CTBR} controller.


\subsubsection{Augmented Geometric Controller}

Finally, the last control mode we implement is an augmentation of this \ac{LVHR} controller, which we refer to as \ac{LVHR} plus gains, or \ac{LVHR}+g for short.
Geometric control can enhance the learning process, as the policy does not need to implicitly learn how differential rotor thrusts generate torques; this mapping is handled by the controller.
However, this approach does not necessarily guarantee better performance, as the policy can be significantly constrained by the tuning of the controller gains.
For this reason, it is of interest to investigate how performance is affected when the policy is allowed to output the controller gains in addition to the velocity and heading rate commands.

Thus, when this control mode is selected, the policy outputs are the same as in \eqref{eq:geo_vel} and \eqref{eq:geo_ang}, but the policy also outputs the controller gains $\{k_v, k_R, k_\omega\}$. 
Recall that the policy outputs are constrained between zero and one, so they are scaled to the minimum and maximum allowed gains selected by the user,

\begin{equation}
    k_v^{\text{cmd}} = k_v^{\text{min}} + u_5 \left(k_v^{\text{max}} - k_v^{\text{min}}\right)
\end{equation}
\begin{equation}
    k_R^{\text{cmd}} = k_R^{\text{min}} + u_6 \left(k_R^{\text{max}} - k_R^{\text{min}}\right)
\end{equation}
\begin{equation}
    k_{\omega_{b_{3}}}^{\text{cmd}} = k_{\omega_{b_{3}}}^{\text{min}} + u_7 \left(k_{\omega_{b_{3}}}^{\text{max}} - k_{\omega_{b_{3}}}^{\text{min}}\right).
\end{equation}


\section{Training Tasks} \label{sec:tasks}

The training objectives considered for this thesis aim to provide an approximate solution to the \ac{OCP}s, formulated in \autoref{sec:simpmin-OCP}.
Unlike traditional solvers that rely on \ac{QP} or model-free \ac{RL}, which treat the system dynamics as a black box, we leverage the simulator's automatic differentiation engine to compute the analytical gradient from the cost function to the policy parameters.

Using this method, to achieve near-optimal results the training environment and policy architecture must be carefully designed.
In our simulation, we formulate them for the trajectory tracking and gate progress tasks in the following manner.

\subsection{Trajectory Tracking}\label{subsec::TT}

For the trajectory tracking task (see \autoref{tt-task}), the objective is to minimize a cumulative loss function over a finite horizon of length $H$, defined with respect to a reference state $\mathbf{x}_k^\text{ref}$ and the quadrotor state $\mathbf{x}_k$.
The stage loss at each timestep $k$, denoted as $\mathcal{L}(\mathbf{x}_k^\text{ref}, \mathbf{x}_k)$, is formulated as a weighted sum of position, velocity, and attitude alignment errors, together with a penalization of the body rates to improve stability during early training,

\begin{equation}\label{acc_loss}
\mathcal{L}(\mathbf{x}_k^\text{ref}, \mathbf{x}_k) =
\lambda_p \|\mathbf{p}_k^\text{ref} - \mathbf{p}_k\|_2 +
\lambda_v \|\mathbf{v}_k^\text{ref} - \mathbf{v}_k\|_2 +
\lambda_a \left(1 - \mathbf{\hat{b}}_{3,k} \cdot \mathbf{z}_k \right) +
\lambda_\omega \|\boldsymbol{\omega}_k\|_2^2
\end{equation}
 where $\lambda_{()}$ are the weighting factors. The desired thrust direction $\mathbf{z}_k$ is defined as,

\begin{equation}
\mathbf{z}_k =
\frac{\mathbf{a}_k^\text{ref} + \mathbf{g}}{\|\mathbf{a}_k^\text{ref} + \mathbf{g}\|_2}.
\end{equation}
Since the simulator is batched, i.e., multiple environments run in parallel, the total cumulative cost over the horizon is defined as,

\begin{equation}\label{cum-cost}
J_k = \frac{1}{B}\frac{1}{H} \sum_{i=1}^{B} \sum_{h=k}^{k+H-1} \mathcal{L}\big(\mathbf{x}_{h,i}^\text{ref}, \mathbf{x}_{h,i}\big)
\end{equation}
where $k$ denotes the initial timestep of the current horizon, $H$ is the horizon length, and $B$ is the number of parallel environments.
At the end of each horizon, the gradient of the cumulative cost in \eqref{cum-cost} with respect to the policy parameters is backpropagated through time for optimization, as described in \autoref{sec:diff-opt}.

To improve numerical stability and avoid exploding gradients, training episodes are terminated when predefined conditions are met.
This prevents the accumulation of large errors, which can otherwise lead to unstable learning dynamics.
For this task, episode termination is based solely on the position-tracking error.
Specifically, an episode is terminated when the euclidean distance between the quadrotor position and the reference trajectory exceeds a predefined threshold $d_{\max}$,
\begin{equation}
d_{\max} < \|\mathbf{p}_k^\text{ref} - \mathbf{p}_k\|_2. 
\end{equation}
If the termination condition is triggered during an episode, the quadrotor state is reset to the reference trajectory state and the loss over that horizon is not taken into account.

Regardless of the training task, the policy network is provided with the full state of the quadrotor.
For the trajectory tracking task, the observation is augmented with reference trajectory information.
Specifically, instead of providing absolute position and velocity, the observation includes tracking errors from the reference trajectory.
This improves learning efficiency by explicitly encoding the control objective.
The observation vector is defined as,
\begin{equation}\label{obs-TT}
\mathbf{o}_k =
\begin{bmatrix}
\mathbf{p}_k^\text{ref} - \mathbf{p}_k & 
\mathbf{v}_k^\text{ref} - \mathbf{v}_k & 
\mathbf{a}_k^\text{ref} &
\mathbf{q}_k &
\boldsymbol{\omega}_k &
\end{bmatrix}^\top
\in \mathbb{R}^{16}.
\end{equation}

The policies are first trained to track arbitrary trajectories created by summing random harmonics as described in \eqref{harmonics}, where $n=3$, $\phi=2$, and $\omega$ is initially $\omega=1$ . 
A curriculum is used so that the trajectory speed $\omega$  is increased by a certain $\delta$ when the \ac{RMSE} falls below a given threshold.
Note that the reference is trajectory is described in the world frame $\mathcal{W}$.

\begin{equation}\label{harmonics}
    \mathbf{p}_k^{\text{ref}} = \sum_{i=1}^{n}A_i\sin{\left(\omega_i k + \phi_i\right)}
\end{equation}

Then, the reference velocity and acceleration are approximated as the discrete derivatives of the position, where $T_s=0.01$.

\begin{equation}
        \mathbf{v}_k^\text{ref} \approx \frac{\mathbf{p}_{k}^\text{ref} - \mathbf{p}_{k-1}^\text{ref}}{T_s}
\end{equation}
\begin{equation}
        \mathbf{a}_k^\text{ref} \approx \frac{\mathbf{v}_{k}^\text{ref} - \mathbf{v}_{k-1}^\text{ref}}{T_s}
\end{equation}

\subsection{Gate Progress}\label{gate-prog}

Similarly, the gate progress loss function, \eqref{loss_gp}, is accumulated over a horizon of length $H$ and over $B$ parallel environments as in \eqref{acc_loss}.
In this case, the loss is a function of the current target gate, the drone's current position and the drone's previous position (see \autoref{fig:gate_geometryy}).
Additionally, we penalize high body rates to encourage stability during early training.

\begin{equation}\label{loss_gp}
\mathcal{L} =  \left\| \mathbf{p}_{k}^{\mathcal{G}}  \right\|_2 - \left\| \mathbf{p}_{k-1}^{\mathcal{G}} \right\|_2 + \lambda_\omega \left\| \boldsymbol{\omega} \right\|_2.
\end{equation}

Note that the gate progress is expressed in the gate frame.
We define the gate plane in such a manner that the $\hat{\mathbf{g}}_2$ axis is the normal and the drones are always behind the gate, i.e., in the negative $\hat{\mathbf{g}}_2$ direction.
Given the gate position and orientation in the world frame, the \ac{UAV}'s relative position with respect to the gate is computed as

\begin{equation}
    \mathbf{p}^{\mathcal{G}} = \mathbf{R}_{\mathcal{G}}^\top \left( \mathbf{p}^\mathcal{W} - \mathbf{p}^\mathcal{W}_{\text{gate}} \right).
\end{equation}

When the $\hat{\mathbf{g}}_2$ component of $\mathbf{p}_{k-1}^\mathcal{G}$ is negative and the $\hat{\mathbf{g}}_2$ component of $\mathbf{p}_{k}^\mathcal{G}$ is nonnegative, the gate plane is considered to be crossed.

    \centering

    \caption{Gate progress task representation.}
    \label{fig:gate_geometryy}
\end{figure}

During the crossing, if condition \eqref{eq:condition} is met, the gate is considered successfully crossed and the gate frame is updated to the next gate of the track.
Otherwise, a gate crash is recorded, the environment is terminated and the drone is initialized either at the starting position or behind a random gate of the track.
We also set boundaries in the world frame; if the drone crosses those boundaries the environment is terminated to save training time.
When environments are terminated, their loss is not accumulated.

\begin{equation}
    \label{eq:condition}
    \quad \quad
    \left| p^\mathcal{G}_{\hat{g}_1} \right| < \frac{w}{2}
    \quad \wedge \quad
    \left| p^\mathcal{G}_{\hat{g}_3} \right| \leq \frac{h}{2}
\end{equation}

The observation vector for this task is,

\end{equation}
where $\mathbf{\hat{g}_2}$ is the gate forward axis expressed in the world frame and $g_\text{idx}$ is the current gate index from the track.

\section{Domain Randomization}\label{sec:DR}
To improve the transferability of the policy and to prevent the it from ovrefitting to a specific environment, \ac{DR} is implemented in our methodology.
A domain can be defined as a specific subset of all possible \ac{POMDP} environments. 
While a complete \ac{POMDP} is formally characterized by an 8-tuple~\cite{ASTROM1965174}, 
we focus on a reduced 6-tuple representation that captures the essential elements 
for our domain definition. Hence, a standard domain is characterized by 
$\mathcal{D} = (\mathcal{S}, \mathcal{A}, \mathcal{O}, P, O, R)$ where:

\begin{itemize}
    \item $\mathbf{\mathcal{S}:}$ denotes a set of states called the state space.
    \item $\mathbf{\mathcal{A}:}$ is the set of actions called the action space.
    \item $\mathbf{\mathcal{O}:}$ is the set of observations called the observation space.
    \item $\mathbf{P(s'|s, a):}$ is the probability that a given action $a \in A$, at state $s \in S$ will lead to a new state $s' \subset S$. 
    \item $\mathbf{O(o|s', a):}$ is the observation probability, representing the probability of observing $o \in \mathcal{O}$ after taking action $a$ and transitioning to state $s'$.
    \item $\mathbf{R(s',s,a):}$ is the immediate reward, or expected inmediate reward, received for transitioning from $s$ to $s'$ after performing action $a$. 
\end{itemize}
From the source domain, $\mathcal{D}_{\xi_s}$, (i.e. simulation), we can control a set of N parameters that define a configuration vector, $\boldsymbol{\xi} = [\xi_i, \xi_{i+1}, ..., \xi_N]$.
These parameters will affect the transition probability $\mathbf{P(s'|s,a)}$ of the domain.
Examples of $\xi_i$ can include:

\begin{tabular}{c|c}

    Physical properties & $\xi_{\text{mass}}, \ \xi_{\text{inertia}}, \ \xi_{\text{drag}}, \ \xi_{\text{gravity}}$ \\
    Perception properties & $\xi_{\text{sensor\_noise}}, \ \xi_{\text{sensor\_latency}}$ \\
    Viual properties & $\xi_{\text{lighting}}, \ \xi_{\text{texture}}, \ \xi_{\text{fov}}$ \\
    Scene properties & $\xi_{\text{gate\_pos}}, \ \xi_{\text{gate\_size}}, , \ \xi_{\text{gate\_num}}$ \\
    
\end{tabular}

\noindent When \ac{DR} is applied, each configuration $\boldsymbol{\xi}$ is sampled from a bounded randomization space $\Xi$ with a probability distribution. 
That is to say, $\boldsymbol{\xi} \in \Xi \subset \mathbb{R}^N$,  $\boldsymbol{\xi} \sim p(\boldsymbol{\xi})$ where $\xi_i \in [\xi_i^{\text{low}}, \xi_i^{\text{high}}]$. 
At the start of every training episode, a new environment insance is created by sampling a new configuration vector $\boldsymbol{\xi}$ from the distribution $p(\boldsymbol{\xi})$. 
Therefore, the policy will interact with a new \ac{POMDP} environment, i.e. a new domain $\mathcal{D}_{\boldsymbol{\xi}}$. 
This way, data from a vareity of environments that otherwise would reamain unexplored can be collected. 

During training, the neural network weights and biases $\theta$ are optimized to minimize the expected loss across a distribution of randomized domains~\cite{weng2019DR}.
With the aim of finding the optimal, $\theta^*$ that will lead to the optimal policy $\pi_{\theta}^*$.
As illustrated in \autoref{domains}, the policy is exposed to a variety of domains to promote generalization and improve robustness, increasing the likelihood of successful transfer to the real environment. 
However, this strategy further reduces the sample efficiency of \ac{RL} and increases training time. 
Therefore, it is essential to properly bound the randomization space $\Xi$, so that it encompasses only the minimum necessary range to include the target domain.

\section{Summary}\label{sec:summary}

To summarize the methodology, we present \autoref{fig:cool-diagram}. 
At each time step, the policy takes an observation and outputs control commands.
Depending on the control abstraction selected by the user, the policy will output linear velocities and heading rate, collective thrust and body rates, or single rotor thrust.
Then, the control commands are transformed to rotor speeds and passed through a first-order transfer function to simulate the motor delay.
Next, the rotor speeds are mapped back to thrust and then to wrench values which are used to integrate the translational and rotational dynamics.
Given the next state, the loss function is evaluated and the gradient from this function to the policy parameters is backpropagated to optimize the network.
Note that during backpropagation, a surrogate model is used where the motor dynamics are skipped.

Hence, the training workflow is as follows. 
First, the user selects the training task, referred to in the code as environment env=[tt,\ gp], where (tt) stands for trajectory tracking and (gp) for gate progress.
The user also selects the control abstraction/mode cm=[srt,\ ctbr,\ lvhr,\ lvhr+g], which defines the control mapping function $g_{\text{cm}}()$ from the policy outputs $\boldsymbol{\pi_\theta}(\mathbf{o})$ to the control inputs $\mathbf{u}$. 
Finally, the user introduces the number of parallel environments $B$, the number of episodes $N$, the number of steps $K$, the optimization horizon $H$, and other hyperparameters such as the learning rate $\alpha$, randomization percentage $p$, termination distance threshold $d_\text{max}$, etc. 

\newpage
Once all these parameters are set, training proceeds as follows:

\begin{algorithm}[H]
\caption{Training procedure for differentiable policy optimization.}
\label{alg:training}
\begin{algorithmic}[1]

\Require Training task $\mathrm{env} \in \{\mathrm{tt},\ \mathrm{gp}\}$,
         control mode $\mathrm{cm} \in \{\mathrm{srt},\ \mathrm{ctbr},\ \mathrm{lvhr},\ \mathrm{lvhr{+}g}\}$,
         batch size $B$, episodes $N$, steps $K$, horizon $H$,
         learning rate $\alpha$, simulation step $\Delta t = 0.01\,\mathrm{s}$, other hyperparams...

\State \textbf{Initialize} policy $\pi_\theta$, controller $g_{\mathrm{cm}}$, optimizer

\For{$n = 1, \dots, N$}
    \State Sample domain parameters $\boldsymbol{\xi}^{(i)} \sim p(\boldsymbol{\xi})$ for $i = 1, \dots, B$ \Comment{Domain randomization}
    \State Sample reference $\mathbf{x}_{0:K}^{\mathrm{ref}} \gets \texttt{getReference}(\mathrm{env})$ \Comment{Task-dependent reference}
    \State $\mathbf{x}_0 \gets \mathbf{x}_0^{\mathrm{ref}}$, \quad $J \gets 0$

    \For{$k = 0, \dots, K-1$}

        \State $\mathbf{o}_k \gets \texttt{getObservation}(\mathrm{env},\ \mathbf{x}_k,\ \mathbf{x}_k^{\mathrm{ref}})$ \Comment{Task-dependent observation}

        \State $\mathbf{u}_k \gets g_{\mathrm{cm}}\!\bigl(\pi_\theta(\mathbf{o}_k)\bigr)$ \Comment{Policy forward pass + control mapping}

        \State $\mathbf{x}_{k+1} \gets \mathbf{x}_k + \Delta t \cdot \mathbf{f}(\mathbf{x}_k,\ \mathbf{u}_k,\ \boldsymbol{\xi})$ \Comment{Euler/RK4 integration through dynamics}

        \State $\mathrm{term}_k \gets \texttt{isTerminated}(\mathrm{env},\ \mathbf{x}_{k+1},\ \mathbf{x}_{k+1}^{\mathrm{ref}})$ \Comment{Task-dependent termination}

        \State $J \mathrel{+}= \dfrac{1}{B}\displaystyle\sum_{i=1}^{B} \mathcal{L}_{\mathrm{env}}\!\bigl(\mathbf{x}_{k+1,i}^{\mathrm{ref}},\ \mathbf{x}_{k+1,i}\bigr)\cdot\mathbf{1}\!\left[\lnot\,\mathrm{term}_{k,i}\right]$

        \State Reset terminated environments: $\mathbf{x}_{k+1,i} \gets \mathbf{x}_{k+1,i}^{\mathrm{ref}}\ $ for all $i$ where $\mathrm{term}_{k,i}$

        \If{$(k + 1) \bmod H = 0$} \Comment{End of truncation window}
            \State $\nabla_\theta J \gets \texttt{backward}\!\left(J / H\right)$ \Comment{BPTT through the computational graph}
            \State $\theta \gets \texttt{Adam.step}(\theta,\ \nabla_\theta J,\ \alpha)$ \Comment{Parameter update}
            \State $\mathbf{x}_{k+1} \gets \mathbf{x}_{k+1}.\texttt{detach}()$ \Comment{Detach state to truncate graph}
            \State $J \gets 0$
        \EndIf

    \EndFor
\EndFor

\State \Return $\theta$
\end{algorithmic}
\end{algorithm}
